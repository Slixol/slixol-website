    ### 1. Create src/app/components/ui/LottieAnimation.tsx

    ```typescript
    "use client";

    import { useRef, useState, useEffect } from "react";
    import { useInView } from "framer-motion";
    import Lottie from "lottie-react";

    interface LottieAnimationProps {
      src: string;        // Public path e.g. "/lottie/case-1.json"
      className?: string;
    }

    export default function LottieAnimation({ src, className }: LottieAnimationProps) {
      const ref = useRef<HTMLDivElement>(null);
      // Preload 800px before entering viewport — consistent with project's lazy pattern
      const isInView = useInView(ref, { once: true, margin: "0px 0px 800px 0px" });
      const [animationData, setAnimationData] = useState<object | null>(null);

      useEffect(() => {
        if (!isInView || animationData) return;
        fetch(src)
          .then((r) => r.json())
          .then((data) => setAnimationData(data))
          .catch(() => {/* silently fail — show nothing instead of error */});
      }, [isInView, src, animationData]);

      return (
        <div ref={ref} className={className}>
          {animationData ? (
            <Lottie
              animationData={animationData}
              loop={true}
              autoplay={true}
              style={{ width: "100%", height: "100%" }}
            />
          ) : (
            // Placeholder while loading — matches ImagePlaceholder appearance
            <div className="w-full h-full rounded-xl bg-dark-card/60" />
          )}
        </div>
      );
    }
    ```

    Key decisions (Claude's discretion per CONTEXT.md):
    - `aspect-square` (1:1) instead of `aspect-[4/3]` — Lottie files are 1080x1080 square
    - `loop={true} autoplay={true}` — 4-second geometric loop, always plays
    - No hover interaction (deferred per CONTEXT.md)
    - Fetch pattern (not static import) — files stay out of JS bundle (CASE-02)
    - Silent error handling — blank area on failure is acceptable over crash
